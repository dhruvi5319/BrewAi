**useEffect on item change** — reset all selections to defaults when `item` changes:
- `selectedSize`: find 'Medium' in `item.options.sizes`; if found use 'Medium', else `item.options.sizes[0]?.label ?? ''`
- `selectedMilk`: `item.options.milks[0] ?? ''`
- `selectedTemp`: `item.options.temperatures[0] ?? ''`

**Real-time price calculation:**
```typescript
const sizeOption = item?.options.sizes.find(s => s.label === selectedSize);
const sizeDelta = sizeOption?.delta ?? 0;
const addonTotal = selectedAddons.reduce((sum, label) => {
  const extra = item?.options.extras.find(e => e.label === label);
  return sum + (extra?.price ?? 0);
}, 0);
const perItemPrice = (item?.basePrice ?? 0) + sizeDelta + addonTotal;
const totalPrice = perItemPrice * quantity;
```

**Rendering structure** (use Modal from ui/index.ts with `isOpen` and `onClose`):

```
<Modal isOpen={isOpen} onClose={onClose} title={item.name} size="lg">
  {/* 1. Size selector — always shown if sizes.length > 0 */}
  <fieldset>
    <legend>Size</legend>
    {item.options.sizes.map(size => (
      <label key={size.label}>
        <input type="radio" checked={selectedSize === size.label} onChange={() => setSelectedSize(size.label)} />
        {size.label} {size.delta > 0 ? `+$${size.delta.toFixed(2)}` : size.delta < 0 ? `-$${Math.abs(size.delta).toFixed(2)}` : '$0.00'}
      </label>
    ))}
  </fieldset>

  {/* 2. Milk selector — only if options.milks.length > 0 */}
  {item.options.milks.length > 0 && (
    <fieldset>
      <legend>Milk</legend>
      {item.options.milks.map(milk => (
        <label key={milk}>
          <input type="radio" checked={selectedMilk === milk} onChange={() => setSelectedMilk(milk)} />
          {milk}
        </label>
      ))}
    </fieldset>
  )}

  {/* 3. Temperature selector — only if temperatures.length > 1; if length === 1, show static label */}
  {item.options.temperatures.length === 1 ? (
    <p>Temperature: {item.options.temperatures[0]}</p>
  ) : item.options.temperatures.length > 1 ? (
    <fieldset>
      <legend>Temperature</legend>
      {item.options.temperatures.map(temp => (
        <label key={temp}>
          <input type="radio" checked={selectedTemp === temp} onChange={() => setSelectedTemp(temp)} />
          {temp}
        </label>
      ))}
    </fieldset>
  ) : null}

  {/* 4. Shot count — ONLY if options.shots is non-null (espresso only) */}
  {item.options.shots !== null && (
    <fieldset>
      <legend>Shots</legend>
      {item.options.shots.map(shot => (
        <label key={shot}>
          <input type="radio" checked={selectedShots === shot} onChange={() => setSelectedShots(shot)} />
          {shot}
        </label>
      ))}
    </fieldset>
  )}

  {/* 5. Extras — only if options.extras.length > 0 */}
  {item.options.extras.length > 0 && (
    <fieldset>
      <legend>Add-ons</legend>
      {item.options.extras.map(extra => (
        <label key={extra.label}>
          <input
            type="checkbox"
            checked={selectedAddons.includes(extra.label)}
            onChange={() => setSelectedAddons(prev =>
              prev.includes(extra.label)
                ? prev.filter(a => a !== extra.label)
                : [...prev, extra.label]
            )}
          />
          {extra.label} +${extra.price.toFixed(2)}
        </label>
      ))}
    </fieldset>
  )}

  {/* 6. Special instructions */}
  <div>
    <label htmlFor="special-instructions">Special Instructions</label>
    <textarea
      id="special-instructions"
      maxLength={200}
      value={specialInstructions}
      onChange={e => setSpecialInstructions(e.target.value)}
      rows={3}
    />
    <span className={specialInstructions.length >= 190 ? 'text-error' : 'text-secondary'}>
      {specialInstructions.length}/200
    </span>
  </div>

  {/* 7. Quantity stepper */}
  <div>
    <button
      aria-label="Decrease quantity"
      disabled={quantity <= 1}
      onClick={() => setQuantity(q => Math.max(1, q - 1))}
      className="min-h-[44px] min-w-[44px] ..."
    >−</button>
    <span>{quantity}</span>
    <button
      aria-label="Increase quantity"
      disabled={quantity >= 10}
      onClick={() => setQuantity(q => Math.min(10, q + 1))}
      className="min-h-[44px] min-w-[44px] ..."
    >+</button>
  </div>

  {/* Footer: price + Add to Cart */}
  <div>
    <div>
      <span>${perItemPrice.toFixed(2)} each</span>
      <span>Total: ${totalPrice.toFixed(2)}</span>
    </div>
    <Button variant="primary" onClick={handleAddToCart}>Add to Cart</Button>
  </div>
</Modal>
```

**handleAddToCart function:**
```typescript
const handleAddToCart = () => {
  if (!item) return;
  const cartItem: CartItem = {
    cartItemId: crypto.randomUUID(),
    menuItemId: item.id,
    name: item.name,
    unitPrice: perItemPrice,
    quantity,
    customizations: {
      size: selectedSize,
      milk: item.options.milks.length > 0 ? selectedMilk : null,
      temperature: item.options.temperatures.length > 0 ? selectedTemp : null,
      shots: item.options.shots !== null ? selectedShots : null,
      addons: selectedAddons,
      specialInstructions,
    },
  };
  addItem(cartItem);
  toast.success(`${item.name} added to cart`);
  onClose();
};
```
