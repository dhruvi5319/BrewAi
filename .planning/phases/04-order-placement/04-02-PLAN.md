**Setup helper** — each test needs a non-empty cart before testing order placement. Reuse the pattern from `e2e/customization-cart.spec.ts` if it exists: navigate to menu, click Customize on a drink, pick defaults, click Add to Cart.

```typescript
import { test, expect, Page } from '@playwright/test';

// Helper: add one item to cart via customization modal and close the modal
async function addItemToCart(page: Page) {
  await page.goto('/');
  // Click first "Customize" button visible on page
  await page.getByRole('button', { name: 'Customize' }).first().click();
  // Wait for customization modal to appear
  await page.getByRole('dialog').waitFor({ state: 'visible' });
  // Default selections are pre-applied; just click Add to Cart
  await page.getByRole('button', { name: 'Add to Cart' }).click();
  // Wait for modal to close and toast to appear
  await page.waitForTimeout(300);
}

// Helper: open cart drawer
async function openCart(page: Page) {
  await page.getByRole('button', { name: /open cart/i }).click();
  await page.getByRole('dialog', { name: /your cart/i }).waitFor({ state: 'visible' });
}
```

**Test 1 — Loading state: Place Order button shows spinner + disabled during submission**
```typescript
test('Place Order button shows loading state during submission', async ({ page }) => {
  await addItemToCart(page);
  await openCart(page);

  // Intercept the POST /api/orders to delay response (so we can see loading state)
  await page.route('**/api/orders', async (route) => {
    await new Promise(resolve => setTimeout(resolve, 1500));
    await route.continue();
  });

  const placeOrderBtn = page.getByRole('button', { name: /place order/i });
  await placeOrderBtn.click();

  // During the delayed request, button should show loading state
  await expect(page.getByText('Placing order…')).toBeVisible();
  await expect(placeOrderBtn).toBeDisabled();
});
```

test('Successful order shows confirmation screen with BRW-NNNNN reference', async ({ page }) => {

});

**Test 5 — Error handling: failed submission shows inline error, cart preserved**
```typescript
test('Failed order submission shows inline error and preserves cart', async ({ page }) => {
  await addItemToCart(page);
  await openCart(page);

  // Intercept POST /api/orders to return a 500 error
  await page.route('**/api/orders', (route) => {
    route.fulfill({
      status: 500,
      contentType: 'application/json',
      body: JSON.stringify({
        data: null,
        error: { code: 'INTERNAL_ERROR', message: 'Something went wrong placing your order. Please try again.' },
        status: 500,
      }),
    });
  });

  await page.getByRole('button', { name: /place order/i }).click();

  // Inline error message must appear (not a toast — inline below button)
  await expect(page.getByText(/something went wrong placing your order/i)).toBeVisible();

  // "Try Again" button must be present
  await expect(page.getByRole('button', { name: /try again/i })).toBeVisible();

  // Cart drawer is still open (user hasn't been navigated away)
  await expect(page.getByRole('dialog', { name: /your cart/i })).toBeVisible();

  // The cart item count should still show items (cart was NOT cleared)
  // At least one item row still visible in cart drawer
  await expect(page.locator('[data-testid="cart-item"]').or(page.locator('text=× 1')).first()).toBeVisible();
});
```
