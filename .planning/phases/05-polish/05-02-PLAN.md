### CustomizationModal

If `src/components/customization/CustomizationModal.tsx` doesn't exist yet, create a complete stub. If it exists, update the layout and animation.

**Responsive layout strategy:** Two layout modes via CSS classes:

- **Mobile (< md):** `fixed bottom-0 left-0 right-0 max-h-[90vh] overflow-y-auto bg-surface-raised rounded-t-2xl z-50`
  - Animation variant: `drawerVariantsBottom` (slides up from bottom)
- **Desktop (md+):** `fixed inset-0 flex items-center justify-center z-50`  
  - Content box: `relative bg-surface-raised rounded-xl max-w-[600px] w-full mx-4 max-h-[90vh] overflow-y-auto`
  - Animation variant: `modalVariants` (scale + fade)

**Implementation pattern:** Rather than rendering two copies, use a SINGLE modal element where the wrapper div uses responsive classes that switch the visual presentation:

```tsx
import { AnimatePresence, motion } from 'framer-motion';
import { modalVariants, drawerVariantsBottom, backdropVariants, useReducedMotion } from '../../lib/motion';

interface CustomizationModalProps {
  isOpen: boolean;
  item: MenuItem | null;
  onClose: () => void;
}

export function CustomizationModal({ isOpen, item, onClose }: CustomizationModalProps) {
  const shouldReduceMotion = useReducedMotion();

  // Escape key handler
  useEffect(() => {
    const handleEscape = (e: KeyboardEvent) => { if (e.key === 'Escape') onClose(); };
    if (isOpen) document.addEventListener('keydown', handleEscape);
    return () => document.removeEventListener('keydown', handleEscape);
  }, [isOpen, onClose]);

  const contentVariants = shouldReduceMotion ? {} : modalVariants;
  const mobileVariants = shouldReduceMotion ? {} : drawerVariantsBottom;
  const bkdropVariants = shouldReduceMotion ? {} : backdropVariants;

  return (
    <AnimatePresence>
      {isOpen && item && (
        <>
          {/* Backdrop */}
          <motion.div
            className="fixed inset-0 bg-canvas/80 backdrop-blur-sm z-50"
            variants={bkdropVariants}
            initial="hidden" animate="visible" exit="exit"
            onClick={onClose}
          />

          {/* Mobile bottom sheet (shown below md, hidden at md+) */}
          <motion.div
            className="md:hidden fixed bottom-0 left-0 right-0 max-h-[90vh] overflow-y-auto bg-surface-raised rounded-t-2xl z-50 pb-safe"
            variants={mobileVariants}
            initial="hidden" animate="visible" exit="exit"
            role="dialog" aria-modal="true" aria-labelledby="modal-title"
          >
            <ModalContent item={item} onClose={onClose} />
          </motion.div>

          {/* Desktop centered dialog (hidden below md, shown at md+) */}
          <div className="hidden md:flex fixed inset-0 items-center justify-center z-50 px-4">
            <motion.div
              className="relative bg-surface-raised rounded-xl max-w-[600px] w-full max-h-[90vh] overflow-y-auto"
              variants={contentVariants}
              initial="hidden" animate="visible" exit="exit"
              role="dialog" aria-modal="true" aria-labelledby="modal-title"
            >
              <ModalContent item={item} onClose={onClose} />
            </motion.div>
          </div>
        </>
      )}
    </AnimatePresence>
  );
}
```

**ModalContent subcomponent** (within same file): Contains the drink name, all customization controls (sizes, milk, temperature, shots, extras, special instructions), quantity stepper, real-time price display, and Add to Cart button.

### CartBadge

Update `src/components/cart/CartBadge.tsx` to add the pop animation on `totalCount` change.

From the existing `badgePopVariants` in motion.ts:
```typescript
export const badgePopVariants: Variants = {
  initial: { scale: 1 },
  pop:     { scale: [1, 1.3, 1], transition: { duration: 0.3, times: [0, 0.5, 1] } },
};
```

Implementation:
```tsx
import { useEffect, useRef } from 'react';
import { motion, useAnimation } from 'framer-motion';
import { badgePopVariants, useReducedMotion } from '../../lib/motion';
import { useCartStore } from '../../stores/cartStore';

export function CartBadge() {
  const { totalCount, openCart } = useCartStore();
  const shouldReduceMotion = useReducedMotion();
  const controls = useAnimation();
  const prevCount = useRef(totalCount);

  useEffect(() => {
    if (totalCount > prevCount.current && !shouldReduceMotion) {
      controls.start('pop');
    }
    prevCount.current = totalCount;
  }, [totalCount, controls, shouldReduceMotion]);

  return (
    <button
      onClick={openCart}
      aria-label={`Open cart, ${totalCount} item${totalCount !== 1 ? 's' : ''}`}
      className="relative min-h-[44px] min-w-[44px] flex items-center justify-center text-primary hover:text-accent transition-colors duration-150 focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-accent focus-visible:ring-offset-2 focus-visible:ring-offset-canvas rounded"
    >
      {/* Cart icon — use Lucide ShoppingCart if available, else emoji */}
      <span aria-hidden="true" className="text-xl">🛒</span>
      {totalCount > 0 && (
        <motion.span
          animate={controls}
          variants={badgePopVariants}
          initial="initial"
          className="absolute -top-1 -right-1 min-w-[20px] h-5 px-1 bg-accent text-canvas text-xs font-bold rounded-full flex items-center justify-center"
        >
          {totalCount}
        </motion.span>
      )}
    </button>
  );
}
```

**Note on Lucide icons:** If `lucide-react` is installed (check package.json), use `<ShoppingCart size={22} />` from `lucide-react` inside the button instead of the emoji. If not installed, keep the emoji placeholder.
